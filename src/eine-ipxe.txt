#!ipxe

prompt --key 0x02 --timeout 3000 Press Ctrl-B for the iPXE command line... && shell ||

set idx:int32 0
set domain yandex.net yandex.ru
set dns6 2a02:6b8::1:1

echo [Eine img/iso] Trying to boot via Netboot...

:loop isset ${net${idx}/mac} || goto fail
   echo Try setup net${idx}
   ifconf net${idx} && chain http://netboot.yandex.net/tftpboot/config.ipxe?mac=${net${idx}/mac} ||
   inc idx && goto loop

:fail

echo Autoconfiguration has failed

prompt --key 0x02 --timeout 3000 Press Ctrl-B for the iPXE command line... && shell ||

sanboot --no-describe --drive 0x80 ||
exit
